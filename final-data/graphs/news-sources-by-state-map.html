import pandas as pd
import matplotlib.pyplot as plt
import numpy as np
# ==============================
# 1️⃣ LOAD FILES
# ==============================
master = pd.read_csv("joint_news_sources(in) (1).csv")
sources = pd.read_csv("combined_outputs.csv")
# ==============================
# 2️⃣ EXTRACT DOMAIN FROM URL
# ==============================
# Clean master file
master['domain'] = (
    master['URL']
    .str.lower()
    .str.replace(r'^https?://', '', regex=True)
    .str.replace(r'^www\.', '', regex=True)
    .str.split('/')
    .str[0]
)
# Clean sources file (NEW structure uses 'url')
sources['domain'] = (
    sources['url']
    .str.lower()
    .str.replace(r'^https?://', '', regex=True)
    .str.replace(r'^www\.', '', regex=True)
    .str.split('/')
    .str[0]
)
# ==============================
# 3️⃣ MERGE IDEOLOGY
# ==============================
merged = sources.merge(
    master[['domain', 'ideology']],
    on='domain',
    how='left'
)
# Drop rows without ideology score
merged = merged.dropna(subset=['ideology'])
# ==============================
# 4️⃣ BIN IDEOLOGY SCORES
# ==============================
bins = [-1, -0.1, 0.1, 1]
labels = [
    'Left-leaning (< -0.1)',
    'Center (-0.1 to 0.1)',
    'Right-leaning (> 0.1)'
]
merged['category'] = pd.cut(
    merged['ideology'],
    bins=bins,
    labels=labels
)
# ==============================
# 5️⃣ HISTOGRAM (US-WIDE)
# ==============================
plt.figure(figsize=(10,6))
plt.hist(
    merged['ideology'],
    bins=np.linspace(-1,1,20),
    edgecolor='black'
)
plt.axvline(0, linestyle='--')
plt.title("Distribution of Ideology Scores Across US Newspaper Sources")
plt.xlabel("Political Leaning Score")
plt.ylabel("Number of Articles")
plt.tight_layout()
plt.show()
# ==============================
# 6️⃣ SUMMARY STATISTICS
# ==============================
counts = (
    merged['category']
    .value_counts()
    .reindex(labels)
    .fillna(0)
    .astype(int)
)
mean_score = merged['ideology'].mean()
median_score = merged['ideology'].median()
print("\nCounts by Category:")
print(counts)
print("\nMean Ideology Score:", round(mean_score, 3))
print("Median Ideology Score:", round(median_score, 3))
print("\nTotal Articles with Ideology:", len(merged))

import pandas as pd
import plotly.express as px
# === Load files ===
master = pd.read_csv("joint_news_sources(in) (1).csv")  # old master with ideology
new_data = pd.read_csv("combined_outputs.csv")  # new dataset with gpt/llama/qwen sources
# === Create a unified source-state DataFrame from the new file ===
sources_list = []
for model_prefix in ["gpt", "llama", "qwen"]:
    temp = new_data[[f"{model_prefix}_state", f"{model_prefix}_source"]].copy()
    temp = temp.rename(columns={f"{model_prefix}_state": "state",
                                f"{model_prefix}_source": "source_name"})
    temp = temp.dropna(subset=["state","source_name"])
    sources_list.append(temp)
sources_df = pd.concat(sources_list, ignore_index=True).drop_duplicates()
# === Merge ideology from master file ===
sources_merged = sources_df.merge(
    master[['name','state','ideology']],
    left_on=['source_name','state'],
    right_on=['name','state'],
    how='left'
)
# === Aggregate at state level ===
state_summary = sources_merged.groupby("state").agg(
    count_sources=("source_name", "count"),
    avg_ideology=("ideology", "mean")
).reset_index()
# === 2020 Census population (in 1000s) ===
pop_data = {

    'NM': 2118, 'NY': 20201, 'NC': 10488, 'ND': 779, 'OH': 11799, 'OK': 3981,
    'OR': 4237, 'PA': 13002, 'RI': 1097, 'SC': 5202, 'SD': 887, 'TN': 6911,
    'TX': 29145, 'UT': 3272, 'VT': 643, 'VA': 8631, 'WA': 7705, 'WV': 1794,
    'WI': 5894, 'WY': 576
}
pop_df = pd.DataFrame(list(pop_data.items()), columns=["state","population_1000s"])
# === Merge population and compute sources per million ===
state_summary = state_summary.merge(pop_df, on="state", how="left")
state_summary["sources_per_million"] = state_summary["count_sources"] / (state_summary["population_1000s"] / 1000)
# === Show lowest coverage states ===
low_coverage = state_summary.sort_values("sources_per_million")
print("\nStates with lowest coverage (sources per million residents):")
print(low_coverage[["state","count_sources","population_1000s","sources_per_million"]].head(10))
# === US map visualization ===
fig = px.choropleth(
    state_summary,
    locations="state",
    locationmode="USA-states",
    scope="usa",
    color="avg_ideology",
    hover_data=["count_sources","population_1000s","sources_per_million"],
    color_continuous_scale="RdBu",
    title="US News Sources by State: Count, Ideology, and Coverage"
)
fig.show()

# ============================

    "AL","AK","AZ","AR","CA","CO","CT","DE","FL","GA","HI","ID","IL","IN","IA","KS",
